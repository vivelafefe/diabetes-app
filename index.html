#Clusters plotting function
def plot_current_cluster(Interval,bID,BG1,DV,SC,Sport,Temp,Humid):  
   
    #Concatenate data base and input
    work_df = df[['Interval','bID','BG1','DV','SC','Sport','Temp','Humid','Y-shID']]
    input_df = pd.DataFrame({'Interval': Interval, 'bID': bID, 'BG1': BG1, 'DV': DV, 'SC': SC,
                           'Sport': Sport, 'Temp': Temp, 'Humid': Humid}, index=[0])
    work_df = pd.concat([work_df[:-1],input_df])
    work_df.reset_index(drop=True,inplace=True)
    
    #Prepare data for clustering
    work_df['Sport']=work_df['Sport'].astype(str)
    df_wd = pd.get_dummies(work_df)
    scaler = MinMaxScaler()
    scaled_data = scaler.fit_transform(df_wd)
    scaled_df = pd.DataFrame(scaled_data,columns=df_wd.columns)
    scaled_X = scaled_df.drop('Y-shID',axis=1)
    
    #Clustering
    model = AgglomerativeClustering(n_clusters=None,linkage='complete',distance_threshold=0.5)
    cluster_labels = model.fit_predict(scaled_X)
    current_cluster_indices = [i for i, x in enumerate(cluster_labels) if x == cluster_labels[-1]]
    
    #Check if the cluster is empty, else predict and plot
    scaled_X_cluster = scaled_df.iloc[current_cluster_indices[:-1],[1,2,3,4,5]]
    
    if scaled_X_cluster.empty:
        return pn.pane.Markdown('#_No matching cluster found!',sizing_mode='scale_both',margin=(30,30))
    else:
        #Prepare data for LinearRegression
        scaled_X_pred = scaled_df.iloc[[current_cluster_indices[-1]],[1,2,3,4,5]]
        scaled_y_cluster = scaled_df.iloc[current_cluster_indices[:-1],[6]]
        
        #Predict y with Linear Regression
        model = LinearRegression()
        model.fit(scaled_X_cluster,scaled_y_cluster)
        scaled_y_pred = model.predict(scaled_X_pred) 
        
        #Prepare data for plotting
        #Scaled cluster plot
        scaled_df.iloc[-1,6] = scaled_y_pred
        scaled_cluster_plot = scaled_df.iloc[current_cluster_indices]
        scaled_cluster_plot = scaled_cluster_plot.iloc[:,[1,2,3,4,5,6]]
        #Cluster plot
        inv_data = scaler.inverse_transform(scaled_df.iloc[current_cluster_indices]) 
        cluster_plot = pd.DataFrame(inv_data,columns=scaled_df.columns)
        cluster_plot = cluster_plot.iloc[:,[1,2,3,4,5,6]]
        y_pred = round(cluster_plot.iloc[-1,-1],2)
        #Plotting
        #Tab1
        #List of column names
        cols = cluster_plot.columns
        #Figure
        fig1 = plt.figure(figsize=(13,6.75))
        plt.xlim(0,5)
        plt.ylim(0,1.2)
        plt.grid(linewidth=2)
        #Xticks as column names
        xticks = plt.gca().get_xticks()      
        plt.xticks(xticks,cols)
        #Yticklabels
        yticks = plt.gca().get_yticks()
        start = work_df[cols[0]].min()
        end = work_df[cols[0]].max()
        step = (work_df[cols[0]].max()-work_df[cols[0]].min())/5
        ylabels = np.arange(start,end+step*2,step)
        ylabels = [ '%.2f' % elem for elem in ylabels ]
        plt.gca().set_yticks(yticks,[])
        plt.gca().set_yticklabels(ylabels,va='bottom')
        #Twin Y axes and yticklabels
        for i in range(1,len(cols)):
            ax = plt.twinx()
            ax.spines["left"].set_position(("axes", i/(len(cols)-1)))  
            ax.set_ylim(0,1.2)
            yticks = ax.get_yticks()
            ax.set_yticks(yticks,[])
            start = work_df[cols[i]].min()
            end = work_df[cols[i]].max()
            step = (work_df[cols[i]].max()-work_df[cols[i]].min())/5
            ylabels = np.arange(start,end+step*2,step)
            ylabels = [ '%.2f' % elem for elem in ylabels ]
            ax.set_yticklabels(ylabels,va='bottom')
        plt.gca().yaxis.set_label_position('left')
        plt.gca().yaxis.set_ticks_position('left')
        #Plot lines
        x = range(len(cols))
        #Plot cluster lines
        for i in range(0,len(scaled_cluster_plot[:-1])):    
            y = scaled_cluster_plot.iloc[i]
            plt.plot(x,y,color=pink,linewidth=1)
        #Plot input line   
        y = scaled_cluster_plot.iloc[-1]
        plt.plot(x,y,color=red,linewidth=3)   
        #Annotate input line
        xlist=plt.gca().lines[-1].get_data()[0]
        ylist=plt.gca().lines[-1].get_data()[1]
        for x,y in zip(xlist,ylist):
            plt.text(x+0.03,y+0.03,'%.2f' % cluster_plot.iloc[-1,x],fontweight='bold',fontsize=14)
        #Predicted vs calculated value text
        y_calc = round((input_df.iloc[0,2]-4.7)/2.8 + (input_df.iloc[0,3]*1),2)
        plt.text(3.75,-0.25,f'Predicted ID: {y_pred}',color='white',bbox=dict(facecolor=blue, edgecolor='none', pad=10),
                 ha='right',fontsize=14,fontweight='bold')
        plt.text(4.95,-0.25,f'Calculated ID*: {y_calc}',color='white',bbox=dict(facecolor=green, edgecolor='none', pad=10),
                 ha='right',fontsize=14,fontweight='bold')
        plt.text(5,-0.35,'*Variables used for calculation: correction factor: 2,8 mmol/l, insulin to carb ratio: 1:1',ha='right',color='#8C9496')
        
            
        #Close figure   
        plt.tight_layout()
        plt.close(fig1)
        
        #Tab 2 
        table_df1 = df.iloc[current_cluster_indices[:-1],[0,3,4,5,6,8,9,10,11]]
        table1 = pn.widgets.Tabulator(table_df1,selectable=False,show_index=False,disabled=True,
                                      sizing_mode='scale_both',layout='fit_columns',text_align='left',height=280)
        table_df2 = cluster_plot.iloc[[-1]]
        table_df2.insert(0,'Date',df.iloc[-1,0])
        table_df2.insert(2,'BG2','?')
        table_df2.insert(7,'shID','?') 
        table2 = pn.widgets.Tabulator(table_df2,selectable=False,show_index=False,disabled=True,
                                      sizing_mode='scale_both',layout='fit_columns',text_align='left')
        
        #Tab3
        fig2,axes = plt.subplots(nrows=1,ncols=5,figsize=(15,7))
        axes[0].set_ylabel('Y-shID') 
        for i,col in enumerate(cols[:-1]): 
            axes[i].axhline(y=cluster_plot.iloc[-1,-1], color=blue, linestyle='-',linewidth=1) 
            axes[i].axvline(x=cluster_plot.iloc[-1,i], color=blue, linestyle='-',linewidth=1)
            axes[i].scatter(cluster_plot.iloc[:-1,i],cluster_plot.iloc[:-1,-1],color=pink,s=80)
            axes[i].scatter(cluster_plot.iloc[-1,i],cluster_plot.iloc[-1,-1],color=red,s=120,zorder=2.5)
            axes[i].set_xlabel(col)    
            axes[i].grid(linewidth=2)
            axes[i].text(cluster_plot.iloc[-1,i],cluster_plot.iloc[-1,-1]+0.1,
                         f'{round(cluster_plot.iloc[-1,i],2)}',ha='center',
                         fontweight='bold',fontsize=16)
            
        plt.tight_layout()
        plt.close(fig2)
        
        return pn.Tabs(('Parallel coordinates plot',pn.Column(pn.pane.Markdown('###/ / / Cluster data vs input data'),
                                          pn.pane.Matplotlib(fig1,sizing_mode='scale_both'),
                                          margin=(30,30,0,30),sizing_mode='scale_both')),
                       ('Spread sheet',pn.Column(pn.pane.Markdown('###/ / / Cluster data'),
                                          table1,
                                          pn.Spacer(height=10),
                                          pn.pane.Markdown('###/ / / Input data'),
                                          table2,
                                          margin=(30,30,0,30),sizing_mode='scale_both')),
                       ('Scatter plots',pn.Column(pn.pane.Markdown('###/ / / Cluster vs input'),
                                            pn.pane.Matplotlib(fig2,sizing_mode='scale_both'),
                                            margin=(30,30,0,30),sizing_mode='scale_both')),
                         sizing_mode='scale_both',background=background)
